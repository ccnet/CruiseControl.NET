Dashboard Configuration Block¶

The root tag in the Dashboard Configuration file is the <dashboard> tag.

The <dashboard> tag contains a <remoteServices> section and a <plugins> section.

Example:

 1<dashboard>
 2    <remoteServices>
 3      <server name="linuxBuilder" url="tcp://Metroplex:21234/CruiseManager.rem"  />
 4      <server name="macBuilder" url="tcp://Trypticon:21234/CruiseManager.rem"  />
 5      <server name="winBuilder" url="tcp://OrionPax:21234/CruiseManager.rem"  />
 6    </remoteServices>
 7
 8    <plugins>
 9      <farmPlugins>
10         <farmReportFarmPlugin categories="false" />
11         <cctrayDownloadPlugin />
12         <administrationPlugin password="" />
13      </farmPlugins>
14
15      <serverPlugins>
16         <serverReportServerPlugin />
17      </serverPlugins>
18
19      <projectPlugins>
20         <projectReportProjectPlugin />
21         <viewProjectStatusPlugin />
22         <latestBuildReportProjectPlugin />
23         <viewAllBuildsProjectPlugin />
24      </projectPlugins>
25
26      <buildPlugins>
27         <buildReportBuildPlugin>
28            <xslFileNames>
29               <xslFile>xsl\header.xsl</xslFile>
30               <xslFile>xsl\modifications.xsl</xslFile>
31            </xslFileNames>
32         </buildReportBuildPlugin>
33         <buildLogBuildPlugin />
34      </buildPlugins>
35
36      <securityPlugins>
37         <simpleSecurity />
38      </securityPlugins>
39    </plugins>
40</dashboard>
